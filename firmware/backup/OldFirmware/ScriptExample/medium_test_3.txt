FUNC(PRECISION_MOVE) {
  GROUP(X(180,d1200), Y(90,d1000));
  Z(20,d1500);
  GROUP(T(15,d400), G(80,d500));
  X(220,d800);
  GROUP(Y(110,d600), Z(15,d1000));
  GROUP(X(250,d700), T(30,d500));
}

FUNC(FINE_ADJUSTMENT) {
  GROUP(Y(130,d800), Z(18,d1200));
  GROUP(T(22,d400), G(90,d600));
  X(260,d500);
  GROUP(Y(135,d400), Z(16,d800));
  T(25,d300);
}

X(150,d1000);
Y(80,d800);
CALL(PRECISION_MOVE);
GROUP(X(240,d600), Y(120,d500));
CALL(FINE_ADJUSTMENT);
GROUP(X(280,d1000), Y(140,d800), Z(50,d1200));

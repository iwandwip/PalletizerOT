FUNC(DANCE_SEQUENCE_A) {

  Z(30,d1200);
  GROUP(T(45,d600), G(150,d500));
  GROUP(X(300,d800), Y(150,d600), Z(10,d1000));
  T(75,d700);
}

  X(600,d1000);
  GROUP(Y(300,d800), T(120,d500));
  GROUP(Z(80,d1200), X(550,d800));
}

FUNC(LOAD_PART) {
  X(120,d800);
  Y(60,d600);
  Z(15,d1000);
  GROUP(T(30,d500), G(100,d600));
  GROUP(X(180,d700), Y(90,d500));
}

FUNC(PROCESS_PART) {
  GROUP(Y(140,d900), Z(25,d800));
  GROUP(X(250,d1200), T(60,d600));
  Z(5,d1500);
  GROUP(X(300,d1000), Y(170,d800));
}

FUNC(UNLOAD_PART) {
  GROUP(Z(80,d1200), T(90,d800));
  GROUP(X(400,d1500), Y(200,d1200));
  GROUP(G(50,d600), Z(120,d1000));
}

CALL(LOAD_PART);
CALL(PROCESS_PART);
CALL(UNLOAD_PART);
CALL(LOAD_PART);
CALL(PROCESS_PART);
